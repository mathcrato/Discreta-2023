\item %elon
Sejam $A,B\subset E$. 
\begin{enumerate}
\item Prove que $A\cap B=\varnothing $ se, somente se, $A\subset B^c$.
\item Prove que $A\cup B=E $ se, somente se, $A^c\subset B$.
\end{enumerate}

\Sol{
\begin{enumerate}
		\item $A\cap B=\varnothing \Rightarrow A\subset B^c$ 

				Seja $x\in A$, dado que $A\cap B=\varnothing $ temos que $x\not\in B$ logo $x\in B^c$.
		
		Agora, $ A\subset B^c \Rightarrow A\cap B=\varnothing$ 
		Suponha, por absurdo que $x\in A\cap B$. Assim, $x\in A$ e $x\in B$, dado que $ A\subset B^c$ temos que, como $x\in A \Rightarrow x\in B^c \Rightarrow x\not\in B$ absurdo.
\end{enumerate}
}
