\Sol{
		\begin{enumerate}
		\item \begin{eqnarray*}
		(X \Delta Y) \cap (X \cap Y) &=& [(X-Y)\cup(Y-X)]\cap (X\cap Y)\\
		                             &=& [(X-Y)\cap (X\cap Y)]\cup [(Y-X)\cap (X\cap Y)]\\
		                           &=& \varnothing \cup \varnothing \\
		                           &=& \varnothing
		\end{eqnarray*}
		\end{enumerate}
}
