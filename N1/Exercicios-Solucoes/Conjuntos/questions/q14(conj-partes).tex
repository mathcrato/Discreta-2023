\item  Seja $E = \{\Delta\}$ . Determine $\mathcal{P}(\mathcal{P}(E))$.
	
		\Sol{
				Temos que: 
				\begin{eqnarray*}
						\mathcal{P}(E)=\{\varnothing , \{\Delta \}  \}\\
				\mathcal{P}(\mathcal{P}(E))=\{\varnothing ,  \{\varnothing \}, \{\{\Delta \}\} , \{\varnothing,\{\Delta \} \} \}
				\end{eqnarray*}
		}
