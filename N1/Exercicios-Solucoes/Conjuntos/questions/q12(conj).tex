\item Se um conjunto $A$ tem $1024$ subconjuntos, então o cardinal de $A$, ou número de elementos de $A$, é:
  \begin{multicols}{5}
	\begin{enumerate}
\item $5$
\item $6$
\item $7$
\item $9$
\item $10$
	\end{enumerate}
  \end{multicols}

  \Sol{
		  \begin{eqnarray*}
				  2^n &= & 1024\\
				  2^n &= & 2^{10}\ \Rightarrow 	  n=10
		  \end{eqnarray*}
  }
