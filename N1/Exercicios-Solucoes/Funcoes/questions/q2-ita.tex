\item  A função $f:\mathbb{R}\rightarrow \mathbb{R}$  tal que
  $f(x+y)=f(x)+f(y)$, para todo $x$ e $y$. Calcule $f(0)+1$.

  \Sol{
		  Se $x=y=0$:
		  \begin{eqnarray*}
				  f(0+0)= f(0)+f(0) \Rightarrow 
				  f(0)= f(0)+f(0) \Rightarrow 
				  f(0)= 0  
		  \end{eqnarray*}

		  Potanto, 
		  \begin{eqnarray*}
				  f(0)+1 =0+1=1
		  \end{eqnarray*}
  }
